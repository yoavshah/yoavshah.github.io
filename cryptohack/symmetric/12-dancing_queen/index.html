Challenge

In this challenge we get a python code named “chacha20.py” and the output of the code “output.txt”. In the “chacha20.py” code the programmer created a randomized unknown key, randomized known iv1 (from the output.txt file), randomized iv2(from the output.txt file).

The programmer also encrypt the flag using iv1 and the key, we have access to enc_flag.

And also we have access to msg and enc_msg.

#!/usr/bin/env python3

from os import urandom


FLAG = b'crypto{?????????????????????????????}'


def bytes_to_words(b):
    return [int.from_bytes(b[i:i+4], 'little') for i in range(0, len(b), 4)]

def rotate(x, n):
    return ((x << n) & 0xffffffff) | ((x >> (32 - n)) & 0xffffffff)

def word(x):
    return x % (2 ** 32)

def words_to_bytes(w):
    return b''.join([i.to_bytes(4, 'little') for i in w])

def xor(a, b):
    return b''.join([bytes([x ^ y]) for x, y in zip(a, b)])


class ChaCha20:
    def __init__(self):
        self._state = []

    def _inner_block(self, state):
        self._quarter_round(state, 0, 4, 8, 12)
        self._quarter_round(state, 1, 5, 9, 13)
        self._quarter_round(state, 2, 6, 10, 14)
        self._quarter_round(state, 3, 7, 11, 15)
        self._quarter_round(state, 0, 5, 10, 15)
        self._quarter_round(state, 1, 6, 11, 12)
        self._quarter_round(state, 2, 7, 8, 13)
        self._quarter_round(state, 3, 4, 9, 14)

    def _quarter_round(self, x, a, b, c, d):
        x[a] = word(x[a] + x[b]); x[d] ^= x[a]; x[d] = rotate(x[d], 16)
        x[c] = word(x[c] + x[d]); x[b] ^= x[c]; x[b] = rotate(x[b], 12)
        x[a] = word(x[a] + x[b]); x[d] ^= x[a]; x[d] = rotate(x[d], 8)
        x[c] = word(x[c] + x[d]); x[b] ^= x[c]; x[b] = rotate(x[b], 7)
    
    def _setup_state(self, key, iv):
        self._state = [0x61707865, 0x3320646e, 0x79622d32, 0x6b206574]
        self._state.extend(bytes_to_words(key))
        self._state.append(self._counter)
        self._state.extend(bytes_to_words(iv))

    def decrypt(self, c, key, iv):
        return self.encrypt(c, key, iv)

    def encrypt(self, m, key, iv):
        c = b''
        self._counter = 1

        for i in range(0, len(m), 64):
            self._setup_state(key, iv)
            for j in range(10):
                self._inner_block(self._state)
            c += xor(m[i:i+64], words_to_bytes(self._state))

            self._counter += 1
        
        return c
    

if __name__ == '__main__':
    msg = b'Lorem ipsum dolor sit amet, consectetuer adipiscing elit. Aenean commodo ligula.'
    key = urandom(32)
    iv1 = urandom(12)
    iv2 = urandom(12)

    c = ChaCha20()
    msg_enc = c.encrypt(msg, key, iv1)
    flag_enc = c.encrypt(FLAG, key, iv2)

    print(f"iv1 = '{iv1.hex()}'")
    print(f"iv2 = '{iv2.hex()}'")
    print(f"msg_enc = '{msg_enc.hex()}'")
    print(f"flag_enc = '{flag_enc.hex()}'")

Solution

We will try to rebuild the key from msg, iv2 and enc_msg.

Because of how the encryption routine work, and that it iterate each 64 bytes of the msg, we will just use the first 64 bytes of the message.

In the encryption routine at the end we xor the message with the state, and because we have access to msg and msg_enc, we can calculate the state at the end of the encryption algorithm using xor

c += xor(m[i:i+64], words_to_bytes(self._state))

So to calculate it we use

state_at_end = bytes_to_words(xor(msg, msg_enc))

We can see in the encryption routine it iterate 10 times and execute _inner_block method on the state

for j in range(10):
    self._inner_block(self._state)

So we need to create the inverse method for this method and use it 10 times.

for j in range(10):
    _inner_block_inv(state_at_end)

At the end we get the key from the state from the 4th bytes to the 12th byte, overall 8 bytes. That’s because the first state is built like that

def _setup_state(self, key, iv):
    self._state = [0x61707865, 0x3320646e, 0x79622d32, 0x6b206574]
    self._state.extend(bytes_to_words(key))
    self._state.append(self._counter)
    self._state.extend(bytes_to_words(iv))

So how do we create the inverse function of _inner_block? easy!

_inner_block method is written like that

def _inner_block(self, state):
    self._quarter_round(state, 0, 4, 8, 12)
    self._quarter_round(state, 1, 5, 9, 13)
    self._quarter_round(state, 2, 6, 10, 14)
    self._quarter_round(state, 3, 7, 11, 15)
    self._quarter_round(state, 0, 5, 10, 15)
    self._quarter_round(state, 1, 6, 11, 12)
    self._quarter_round(state, 2, 7, 8, 13)
    self._quarter_round(state, 3, 4, 9, 14)

In order to inverse it we just need to inverse the execution order and instead of _quarter_round create a new method _quarter_round_inv

def _inner_block_inv(state):
    _quarter_round_inv(state, 3, 4, 9, 14)
    _quarter_round_inv(state, 2, 7, 8, 13)
    _quarter_round_inv(state, 1, 6, 11, 12)
    _quarter_round_inv(state, 0, 5, 10, 15)
    _quarter_round_inv(state, 3, 7, 11, 15)
    _quarter_round_inv(state, 2, 6, 10, 14)
    _quarter_round_inv(state, 1, 5, 9, 13)
    _quarter_round_inv(state, 0, 4, 8, 12)

How can we create the _quarter_round_inv? Have a look at _quarter_round

def _quarter_round(self, x, a, b, c, d):
    x[a] = word(x[a] + x[b]); x[d] ^= x[a]; x[d] = rotate(x[d], 16)
    x[c] = word(x[c] + x[d]); x[b] ^= x[c]; x[b] = rotate(x[b], 12)
    x[a] = word(x[a] + x[b]); x[d] ^= x[a]; x[d] = rotate(x[d], 8)
    x[c] = word(x[c] + x[d]); x[b] ^= x[c]; x[b] = rotate(x[b], 7)

All we have to do is to do the operations in inverse order

def _quarter_round_inv(x, a, b, c, d):
    x[b] = rotate(x[b], 32 - 7)
    x[b] ^= x[c]
    x[c] = word(x[c] - x[d])
    x[d] = rotate(x[d], 32 - 8)
    x[d] ^= x[a]
    x[a] = word(x[a] - x[b])
    x[b] = rotate(x[b], 32 - 12)
    x[b] ^= x[c]
    x[c] = word(x[c] - x[d])
    x[d] = rotate(x[d], 32 - 16)
    x[d] ^= x[a]
    x[a] = word(x[a] - x[b])

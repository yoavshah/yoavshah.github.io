Challenge

In this challenge we get a program that creates its own hash algorithm. We need to send two inputs that produce the same hash to gain the flag.

# 2^128 collision protection!
BLOCK_SIZE = 32

# Nothing up my sleeve numbers (ref: Dual_EC_DRBG P-256 coordinates)
W = [0x6b17d1f2, 0xe12c4247, 0xf8bce6e5, 0x63a440f2, 0x77037d81, 0x2deb33a0, 0xf4a13945, 0xd898c296]
X = [0x4fe342e2, 0xfe1a7f9b, 0x8ee7eb4a, 0x7c0f9e16, 0x2bce3357, 0x6b315ece, 0xcbb64068, 0x37bf51f5]
Y = [0xc97445f4, 0x5cdef9f0, 0xd3e05e1e, 0x585fc297, 0x235b82b5, 0xbe8ff3ef, 0xca67c598, 0x52018192]
Z = [0xb28ef557, 0xba31dfcb, 0xdd21ac46, 0xe2a91e3c, 0x304f44cb, 0x87058ada, 0x2cb81515, 0x1e610046]

# Lets work with bytes instead!
W_bytes = b''.join([x.to_bytes(4,'big') for x in W])
X_bytes = b''.join([x.to_bytes(4,'big') for x in X])
Y_bytes = b''.join([x.to_bytes(4,'big') for x in Y])
Z_bytes = b''.join([x.to_bytes(4,'big') for x in Z])

def pad(data):
    padding_len = (BLOCK_SIZE - len(data)) % BLOCK_SIZE
    return data + bytes([padding_len]*padding_len)

def blocks(data):
    return [data[i:(i+BLOCK_SIZE)] for i in range(0,len(data),BLOCK_SIZE)]

def xor(a,b):
    return bytes([x^y for x,y in zip(a,b)])

def rotate_left(data, x):
    x = x % BLOCK_SIZE
    return data[x:] + data[:x]

def rotate_right(data, x):
    x = x % BLOCK_SIZE
    return  data[-x:] + data[:-x]

def scramble_block(block):
    for _ in range(40):
        block = xor(W_bytes, block)
        block = rotate_left(block, 6)
        block = xor(X_bytes, block)
        block = rotate_right(block, 17)
    return block

def cryptohash(msg):
    initial_state = xor(Y_bytes, Z_bytes)
    msg_padded = pad(msg)
    msg_blocks = blocks(msg_padded)
    for i,b in enumerate(msg_blocks):
        mix_in = scramble_block(b)
        for _ in range(i):
            mix_in = rotate_right(mix_in, i+11)
            mix_in = xor(mix_in, X_bytes)
            mix_in = rotate_left(mix_in, i+6)
        initial_state = xor(initial_state,mix_in)
    return initial_state.hex()

Solution

This hash algorithm is actualy really bad. Each block iteration is inversable - meaning that at a given state (current state) and the desired output, we can create a block which will produce it.

First we will create the inverse function scramble_block_inv which is the inverse of scramble_block.

Then we will create the inverse function of the hash cryptohash_specific_block_inv, with a given state (current state) and a desired state it will calculate the needed block.

So now we will create two different blocks, a*32 and b*32, then we will calculate the state of each of them.

$$ S_a = hash(M_a) $$ $$ S_b = hash(M_b) $$

We will choose the desired state to be $0$ block.

So now using the inverse function of the hash we can calculate the needed block to gain the desired hash.

    msg_padded = pad(msg)
    msg_blocks = blocks(msg_padded)

            mix_in = rotate_right(mix_in, i+11)

            mix_in = rotate_left(mix_in, i+6)
        initial_state = xor(initial_state,mix_in)
    return initial_state.hex()


def scramble_block_inv(block):
    for _ in range(40):
        block = rotate_left(block, 17)
        block = xor(X_bytes, block)
        block = rotate_right(block, 6)
        block = xor(W_bytes, block)
        
    return block


def cryptohash_specific_block_inv(curr_state, needed_state, block_index):

    mix_in = xor(curr_state, needed_state)

    for _ in range(block_index):
        mix_in = rotate_right(mix_in, block_index + 6)
        mix_in = xor(mix_in, X_bytes)
        mix_in = rotate_left(mix_in, block_index + 11)

    b = scramble_block_inv(mix_in)
        
    return b




block_a = b"a"*32
block_b = b"b"*32

hashed_block_a = bytes.fromhex(cryptohash(block_a))
hashed_block_b = bytes.fromhex(cryptohash(block_b))

# Now those hashes are the initial states for the next message block
# We need to calculate what the next block needs to be so the overall
# hash will be the same
# We can forge the next block.

next_block_a = cryptohash_specific_block_inv(hashed_block_a, b"\x00"*32, 1)
next_block_b = cryptohash_specific_block_inv(hashed_block_b, b"\x00"*32, 1)


full_block_a = block_a + next_block_a
full_block_b = block_b + next_block_b


print(cryptohash(full_block_a))
print(cryptohash(full_block_b))
